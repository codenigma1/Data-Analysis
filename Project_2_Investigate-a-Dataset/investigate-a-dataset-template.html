In [42]:
# defination of function
def compute(columns_pd):
    
    # combine the datafram
    df_concat = pd.concat(columns_pd, ignore_index=False)
    
    # reset the index
    df_concat.reset_index(inplace=True)
    
    # rename the appropriate column
    df_concat.rename(columns={'index': 'name'}, inplace=True)
    
    # Use group function to get summary of each genre
    df_best = df_concat.groupby('name').sum()
    
    # Sort the value by descending order
    df_best_sort = df_best.sort_values(by='movie_profit', ascending=False)
    
    
    # return the final output
    return df_best_sort

# defination of function
def compute_budget(columns_pd):
    
    # combine the datafram
    df_concat = pd.concat(columns_pd, ignore_index=False)
    
    # reset the index
    df_concat.reset_index(inplace=True)
    
    # rename the appropriate column
    df_concat.rename(columns={'index': 'genre'}, inplace=True)
    
    # Use group function to get summary of each genre
    df_best = df_concat.groupby('genre').sum()
    
    # Sort the value by descending order
    df_best_sort = df_best.sort_values(by='budget_(US_dollar)', ascending=False)
    
    # reset the index
    df_best_sort.reset_index(inplace=True)
    
    # store the result highest and lowest
    highest = df_best_sort.iloc[df_best_sort['budget_(US_dollar)'].idxmax()]
    lowest = df_best_sort.iloc[df_best_sort['budget_(US_dollar)'].idxmin()]
    
    # combine
    final_concat = pd.concat([highest, lowest], axis=1)
    
    # return the final output
    return final_concat
